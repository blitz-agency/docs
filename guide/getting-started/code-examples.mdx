## Base Setup

Every request requires the `x-api-key` header. The client below handles rate limiting, retries on `429`, and error handling.

<Warning>
  **Never expose your API key in client-side code** (browsers, mobile apps). Always call the Blitz API from your backend.
</Warning>

<CodeGroup>

```javascript Node.js / TypeScript
const BLITZ_API_KEY = "YOUR_API_KEY";
const BLITZ_BASE_URL = "https://api.blitz-api.ai";

// Replace 5 with the max_requests_per_seconds value from /v2/account/key-info
function createRateLimiter(maxRps) {
  const timestamps = [];
  let pending = Promise.resolve();

  return () => {
    pending = pending.then(
      () =>
        new Promise((resolve) => {
          const poll = () => {
            const now = Date.now();
            while (timestamps.length && timestamps[0] <= now - 1000)
              timestamps.shift();
            if (timestamps.length < maxRps) {
              timestamps.push(Date.now());
              resolve();
            } else {
              setTimeout(poll, timestamps[0] + 1000 - now + 1);
            }
          };
          poll();
        })
    );
    return pending;
  };
}

const rateLimit = createRateLimiter(5);

async function blitzRequest(method, path, payload) {
  const headers = {
    "x-api-key": BLITZ_API_KEY,
    "Content-Type": "application/json",
  };

  for (let attempt = 0; attempt < 3; attempt++) {
    try {
      await rateLimit();

      const res = await fetch(`${BLITZ_BASE_URL}${path}`, {
        method,
        headers,
        body: payload ? JSON.stringify(payload) : undefined,
      });

      if (res.status === 429) {
        console.log("Rate limited by server, waiting 60s...");
        await new Promise((r) => setTimeout(r, 60000));
        continue;
      }

      if (res.status === 402) {
        console.error("Insufficient credits");
        return null;
      }

      if (res.status === 401) {
        console.error("Invalid API key");
        return null;
      }

      if (res.ok) return await res.json();

      await new Promise((r) => setTimeout(r, 2 ** attempt * 1000));
    } catch (error) {
      if (attempt === 2) {
        console.error(`API error: ${error.message}`);
        return null;
      }
      await new Promise((r) => setTimeout(r, 2000));
    }
  }
  return null;
}
```

```python Python
import time
import threading
from collections import deque
import requests

BLITZ_API_KEY = "YOUR_API_KEY"
BLITZ_BASE_URL = "https://api.blitz-api.ai"
HEADERS = {
    "x-api-key": BLITZ_API_KEY,
    "Content-Type": "application/json",
}

# Replace 5 with max_requests_per_seconds from /v2/account/key-info
def create_rate_limiter(max_rps):
    timestamps = deque()
    lock = threading.Lock()

    def wait():
        with lock:
            while True:
                now = time.monotonic()
                while timestamps and timestamps[0] <= now - 1.0:
                    timestamps.popleft()
                if len(timestamps) < max_rps:
                    timestamps.append(time.monotonic())
                    return
                wait_time = timestamps[0] + 1.0 - now + 0.001
                lock.release()
                try:
                    time.sleep(wait_time)
                finally:
                    lock.acquire()

    return wait

rate_limit = create_rate_limiter(5)

def blitz_request(method, path, payload=None):
    url = f"{BLITZ_BASE_URL}{path}"

    for attempt in range(3):
        try:
            rate_limit()

            if method == "GET":
                response = requests.get(url, headers=HEADERS, timeout=20)
            else:
                response = requests.post(url, headers=HEADERS, json=payload, timeout=20)

            if response.status_code == 429:
                print("Rate limited by server, waiting 60s...")
                time.sleep(60)
                continue

            if response.status_code == 402:
                print("Insufficient credits")
                return None

            if response.status_code == 401:
                print("Invalid API key")
                return None

            if response.status_code == 200:
                return response.json()

            time.sleep(2 ** attempt)

        except Exception as e:
            if attempt == 2:
                print(f"API error: {e}")
                return None
            time.sleep(2)

    return None
```

## Waterfall ICP Search — POST `/v2/search/waterfall-icp-keyword`

Find the best decision-maker at a company using a prioritized cascade. The engine tries each tier in order and fills results from the highest-priority match first.

**Paid plans**: Unlimited. **Free trial**: 1 credit per result. `icp` = cascade tier matched (1 = top priority). `ranking` = overall position in the company (1 = most relevant).

<CodeGroup>

```javascript Node.js
const result = await blitzRequest("POST", "/v2/search/waterfall-icp-keyword", {
  company_linkedin_url: "https://www.linkedin.com/company/openai",
  cascade: [
    {
      include_title: ["CTO", "VP Engineering", "Head of Engineering"],
      exclude_title: ["assistant", "intern", "junior"],
      location: ["WORLD"],
      include_headline_search: false,
    },
    {
      include_title: ["Engineering Director", "Engineering Manager"],
      exclude_title: ["junior", "intern"],
      location: ["WORLD"],
      include_headline_search: false,
    },
    {
      include_title: ["CEO", "Founder", "CTO"],
      exclude_title: ["intern"],
      location: ["WORLD"],
      include_headline_search: false,
    },
  ],
  max_results: 5,
});

for (const person of result.results) {
  const p = person.person;
  console.log(`[Tier ${person.icp} | Rank #${person.ranking}] ${p.full_name} — ${p.headline}`);
  console.log(`  LinkedIn: ${p.linkedin_url}`);
  // icp: which cascade tier matched (1 = top priority)
  // ranking: overall position in company (1 = most relevant)
}
```

## People Search — POST `/v2/search/people`

Find people across multiple companies by combining company AND people filters. Supports cursor-based pagination. See [People Search guide](/guide/concepts/people-search) for full documentation.

**Paid plans**: Unlimited. **Free trial**: 1 credit per result. Pagination: pass the `cursor` from the response to fetch the next page. `cursor: null` means last page.

<Note>
  People Search uses `location.country` for person country filter (not `country_code` as in Employee Finder). The accepted ISO 2-letter codes are identical.
</Note>

<CodeGroup>

```javascript Node.js
const result = await blitzRequest("POST", "/v2/search/people", {
  company: {
    keywords: { include: ["SaaS"] },
    employee_range: ["51-200", "201-500"],
    hq: { country_code: ["US"] },
  },
  people: {
    job_title: { include: ["Sales"] },
    job_level: ["VP", "Director"],
    job_function: ["Sales & Business Development"],
  },
  max_results: 10,
});

console.log(`Found ${result.results_length} results`);
for (const person of result.results) {
  // People Search returns person fields directly (not nested in .person)
  console.log(`${person.full_name} — ${person.headline}`);
  console.log(`  LinkedIn: ${person.linkedin_url}`);
}

// Paginate with cursor
if (result.cursor) {
  const nextPage = await blitzRequest("POST", "/v2/search/people", {
    company: {
      keywords: { include: ["SaaS"] },
      employee_range: ["51-200", "201-500"],
      hq: { country_code: ["US"] },
    },
    people: {
      job_title: { include: ["Sales"] },
      job_level: ["VP", "Director"],
      job_function: ["Sales & Business Development"],
    },
    max_results: 10,
    cursor: result.cursor,
  });
}
```

```python Python
result = blitz_request("POST", "/v2/search/people", {
```
